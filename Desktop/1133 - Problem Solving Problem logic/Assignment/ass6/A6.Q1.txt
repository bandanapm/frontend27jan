//To create a program where user input first name in one arrey and lastname in a parallel arrey. The output should be a list of email addresses where the address is of the following form: first.last@mycollege.edu

//Declare var
Declare firstNames[], firstName as String
Declare lastNames[], lastName as String
Declare sentinel as String
Declare i as Integer

//Set sentinel character
Set sentinel = "exit"

//Checking conditions and output
//Repeating loop until user enter sentinal value
While True 
	//Enter first name 
	Write "Enter first name or exit to end the program: "
	Input firstName

	//Check if the user enter first name or sentinel value
	If firstName == sentinel
		Exit Loop
	End If

	//Enter last name 
	Write "Enter last name: "
	Input lastName

	//Adding first and last name to the fisrt and last name array
	Append firstName To firstNames[]
	Append lastName To lastNames[]

End While

//Output
For i = 1 to Length_Of(firstNames)
    Write "Email address for " + firstNames[i] + " " + lastNames[i] + ": " + firstNames[i] + "." + lastNames[i] + "@mycollege.edu"
End For
