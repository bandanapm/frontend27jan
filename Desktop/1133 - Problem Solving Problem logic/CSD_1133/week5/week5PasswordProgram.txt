//Credential checker, Validate the username and password

//Declare vars
Declare usernameProgram As String
Declare passwordProgram As String
Declare username As String
Declare password As String

//Init vars
Set usernameProgram = "Bob"
Set passwordProgram = "Bob"

//Welcoome Message
Write"Welcome to credential checker 2050"

//Taking input
Write "Please enter your username"
Input username
Write "Please enter your password"
Input password

//Validate username and password
If usernameProgram==username and passwordProgram == password //Validate that all are correct
	Write "Access Granted"
Else If usernameProgram==username or passwordProgram == password //Validate if all are incorrect
	If usernameProgram==username //Validate for which input is incorrect
		Write "Invalid password"
	Else
		Write "Invalid username"
	End If
Else
	Write"Invalid Login"
End If
