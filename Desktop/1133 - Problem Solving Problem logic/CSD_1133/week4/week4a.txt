//Variables used : Age, AgeLimit
//Age checker, This program will check a users age to drive

//Declare var
Declare Age As Integer
Declare AgeLimit As Integer

//Take user input
Write "Please enter your age"
Input Age

//Initialize vars
Set AgeLimit = 16

//Check user age
If Age > AgeLimit Then
	Write "you can drive!"
Else
	Write "Sorry, you're too young"
End If
